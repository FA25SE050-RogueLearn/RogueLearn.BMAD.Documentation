│ │ void bubbleSort(int arr[], int n) {             │ │
│ │     for (int i = 0; i < n-1; i++) {            │ │
│ │         for (int j = 0; j < n-i-1; j++) {      │ │
│ │             if (arr[j] > arr[j+1]) {           │ │
│ │                 // Swap elements               │ │
│ │                 int temp = arr[j];             │ │
│ │                 arr[j] = arr[j+1];             │ │
│ │                 arr[j+1] = temp;               │ │
│ │             }                                   │ │
│ │         }                                       │ │
│ │     }                                           │ │
│ │ }                                               │ │
│ │ ```                                             │ │

│                                                     │

│ │ |-------------|------|---------|-------|       │ │
│ │ | Bubble Sort | O(n) | O(n²)   | O(n²) |      │ │
│ │ | Quick Sort  | O(n log n) | O(n log n) | O(n²) | │ │
│ │ | Merge Sort  | O(n log n) | O(n log n) | O(n log n) | │ │
